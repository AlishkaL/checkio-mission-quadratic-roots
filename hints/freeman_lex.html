Here is the whole solution:

from collections.abc import Iterable
import math


def quadratic_roots(a: int, b: int, c: int) -> Iterable[int|str]:
    
    discriminant = b * b - 4 * a * c
    if discriminant > 0:
        root1 = (-b + math.sqrt(discriminant)) / (2 * a)
        root2 = (-b - math.sqrt(discriminant)) / (2 * a)
        return [root1, root2]
    if not discriminant:
        root = -b / (2 * a)
        return [root, root]
    return ["No real roots"]

In this solution you return an Iterable object, not a list. Use try/except to handle case when discriminant < 0, then yield first root, calculate and yield second root of first root again. 

from collections.abc import Iterable
import math


def quadratic_roots(a: int, b: int, c: int) -> Iterable[int|str]:
    
    try:
        d_sq = math.sqrt(b * b - 4 * a * c)
        yield (root1 := (-b + d_sq) / (2 * a))
        yield d_sq and (-b - d_sq) / (2 * a) or root1
    except: yield "No real roots"
