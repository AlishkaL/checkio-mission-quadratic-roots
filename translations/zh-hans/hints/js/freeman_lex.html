function quadraticRoots(a: number, b: number, c: number): [number, number] | [string] {
    
    const discriminant = b * b - 4 * a * c;
    if (discriminant> 0) {
        const root1 = (-b + Math.sqrt(discriminant)) / (2 * a);
        const root2 = (-b - Math.sqrt(discriminant)) / (2 * a);
        return [root1, root2]}
    if (!discriminant) {
        const root = -b / (2 * a);
        return [root, root]}
    return ["No real roots"]
}
